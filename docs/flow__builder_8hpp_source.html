1719// Function: erase
1720inline void FlowBuilder::erase(Task task) {
1721
1722 if (!task._node) {
1723 return;
1724 }
1725
1726 // remove task from its successors' predecessor list
1727 for(size_t i=0; i<task._node->_num_successors; ++i) {
1728 task._node->_edges[i]->_remove_predecessors(task._node);
1729 }
1730
1731 // remove task from its precedessors' successor list
1732 for(size_t i=task._node->_num_successors; i<task._node->_edges.size(); ++i) {
1733 task._node->_edges[i]->_remove_successors(task._node);
1734 }
1735
1736 _graph._erase(task._node);
1737}

1740// Procedure: _linearize
1741template <typename L>
1742void FlowBuilder::_linearize(L& keys) {
1743
1744 auto itr = keys.begin();
1745 auto end = keys.end();
1746
1747 if(itr == end) {
1748 return;
1749 }
1750
1751 auto nxt = itr;
1752
1753 for(++nxt; nxt != end; ++nxt, ++itr) {
1754 itr->_node->_precede(nxt->_node);
1755 }
1756}
